#include <iostream>
using namespace std;

class Number {
private:
    int value;

public:
    Number(int v = 0) : value(v) {}

    Number operator-() {
        return Number(-value);
    }

  
    Number operator+(const Number &obj) {
        return Number(value + obj.value);
    }

    friend ostream& operator<<(ostream &out, const Number &obj);


    friend istream& operator>>(istream &in, Number &obj);
};


ostream& operator<<(ostream &out, const Number &obj) {
    out << obj.value;
    return out;
}


istream& operator>>(istream &in, Number &obj) {
    in >> obj.value;
    return in;
}

int main() {
    Number n1, n2, result;

    cout << "Enter first number: ";
    cin >> n1;
    cout << "Enter second number: ";
    cin >> n2;

    // Binary operator (+)
    result = n1 + n2;
    cout << "Sum: " << result << endl;

    // Unary operator (-)
    result = -n1;
    cout << "Negation of first number: " << result << endl;

    return 0;
}
